### 큰 따옴표, 작은 따옴표 규칙
```python
# 둘 중에 암거나 써도 상관 없다, 심지어 값, 속성 모두 같다
print('a'is"a") #출력: True

# 따옴표 있는 문자열을 출력하고 싶을 때 둘을 같이 써준다
print("'스투페파이'") #출력: '스투페파이'
print('"봄바르다"') #출력: "봄바르다"

# 따옴표 세개를 써주면 여러 줄의 문자열을 만들 수 있다
print(
    """
        가
        '나'
        "다"
    """
    #출력: 가
    #     '나'
    #     "다"
)
```
